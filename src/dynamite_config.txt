## DYNAMITE Configuration File
## Jeremy Dietrich
## 2020 July 6

# mode: Run mode for DYNAMITE - currently accepts single/Tess/Kepler/all

mode::single

# system: Single system to run on - only used if mode is single

system::tau Ceti

# period: Probability distribution for planet periods - currently accepts epos/syssim

period::epos

# radius: Probability distribution for planet radii - currently accepts epos/syssim

radius::syssim

# radtype: planet radius function type - powerlaw/clustered, radius must be syssim

radtype::clustered

# radmin: planet radius minimum - 0.5 for full Kepler popluation stats, 0.1 for reaching the smallest planets in TOI multi systems

radmin::0.1

# radmax: planet radius maximum - 5 for Neptunes and below, 10 for full Kepler population stats, 15 for largest super-Jupiter planets in TOI multi systems

radmax::5

# plot: run plotting code - True/False; must have at least one individual plotting parameter below also true for any plot to output

plot::True

# saved: use saved values instead of re-running DYNAMITE - switch to true for plotting

saved::False

# plt_P_R: plot the periods and radii of each system, as well as against each other - mode must not be single

plt_P_R::False

# plt_tdtp: plot the transit probability vs the transit depth for all systems - mode must not be single

plt_tdtp::False

# plt_deltas: plot the injection probability compared to the separation stability criterion in terms of mutual Hill radii

plt_deltas::False

# plt_ratios: plot the period ratios of successive pairs of planets across all systems - mode must not be single

plt_ratios::False

# plt_indpars: plot the histograms of the individual parameters of a system - mode must be single

plt_indpars::True
